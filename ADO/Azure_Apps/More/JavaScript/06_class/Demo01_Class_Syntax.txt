// 150	JavaScipt 的函式也是物件（一）

    <script>

        function Box() {
            this.color = 'black';
            this.paint = function (value) {
                this.color = value;
            }
        }

        var objBox = new Box();
        objBox.paint("Blue");
        debug.innerHTML = objBox.color;

    </script>
